
    # 【1】正则匹配路径：拦截带或不带 http(s):// 的 sinaimg.cn 链接
    # 对应 JS: replace(/https:(\/)*/,'').replace(/http:(\/)*/, '') 和 includes("sinaimg.cn")
    location ~* ^/(?:https?:/+|/)?([^/]+\.sinaimg\.cn)(/.*)$ {
        # 提取域名和 URI
        set $sina_domain $1;
        set $sina_uri $2;

        # 动态 proxy_pass 需要配置 DNS 解析器，否则 Nginx 启动会报错
        resolver 8.8.8.8 1.1.1.1 valid=300s;

        # 【2】处理 OPTIONS 预检请求 (解决跨域)
        if ($request_method = 'OPTIONS') {
            add_header 'Access-Control-Allow-Origin' '*' always;
            add_header 'Access-Control-Allow-Methods' 'GET,HEAD,OPTIONS' always;
            add_header 'Access-Control-Allow-Headers' 'Content-Type,Content-Length,Accept-Encoding,X-Requested-with,Origin' always;
            add_header 'Access-Control-Allow-Credentials' 'true' always;
            add_header 'Content-Type' 'text/plain charset=UTF-8';
            add_header 'Content-Length' 0;
            return 204;
        }

        # 【3】伪造回源请求头 (对应 JS 中的 fetch headers)
        proxy_set_header User-Agent "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/139.0.0.0 Safari/537.36 Edg/139.0.0.0";
        proxy_set_header Referer "https://weibo.cn/";
        proxy_set_header Host $sina_domain; # 必须重写 Host，否则新浪服务器会拒绝响应

        # 开启 SNI 支持（极其重要，否则代理到 HTTPS 目标时可能会报 502）
        proxy_ssl_server_name on;
        proxy_ssl_protocols TLSv1.2 TLSv1.3;

        # 【4】执行回源代理
        proxy_pass https://$sina_domain$sina_uri;

        # 【5】配置本地缓存 (对应 JS 中的 cache.match 和 cache.put)

        
        # 隐藏源站自带的可能会引起冲突的响应头
        proxy_hide_header Access-Control-Allow-Origin;
        proxy_hide_header Cache-Control;
        
        # 【6】添加响应给客户端的 Header (对应 JS 中的 resHdrs)
        add_header 'Access-Control-Allow-Origin' '*' always;
        add_header 'Access-Control-Allow-Credentials' 'true' always;
        add_header 'Access-Control-Allow-Headers' 'Content-Type,Content-Length,Accept-Encoding,X-Requested-with,Origin' always;
        add_header 'Access-Control-Allow-Methods' 'GET,HEAD,OPTIONS' always;
        add_header 'Cache-Control' 'public, max-age=31536000' always;
        add_header 'X-Nginx-Cache' $upstream_cache_status always; # 类似于 JS 里的 X-Worker-Cache
    }

    # 【7】兜底逻辑：对于不匹配 *.sinaimg.cn 的请求，直接返回 404 (对应 JS 第一层 404 判断)
    location / {
        return 404;
    }
